### 6. Content Verification Approach

**Automated Content Comparison Script:**
```typescript
// scripts/verify-template-content.ts
/**
 * Compares template content with seed.ts prompts
 * Fails if any mismatch detected
 */
async function verifyTemplateContent() {
  const seedPrompts = extractPromptsFromSeed(); // Parse seed.ts
  const templatePrompts = await getAllTemplates(); // From PromptTemplateRegistry
  
  const mismatches = [];
  for (const [roleName, seedPrompt] of Object.entries(seedPrompts)) {
    const template = templatePrompts.get(roleName);
    if (!template || template.content !== seedPrompt) {
      mismatches.push({ roleName, expected: seedPrompt, actual: template?.content });
    }
  }
  
  if (mismatches.length > 0) {
    console.error('Template content mismatches detected:', mismatches);
    process.exit(1);
  }
  
  console.log('✓ All templates match seed.ts');
}
```
