```typescript
class TemplateDeploymentCircuitBreaker {
  private failureThreshold = 5;
  private failureCount = 0;
  private state: 'CLOSED' | 'OPEN' | 'HALF_OPEN' = 'CLOSED';
  
  async deployTemplate(template: PromptTemplate): Promise<DeploymentResult> {
    if (this.state === 'OPEN') {
      return this.rollbackToPreviousVersion();
    }
    
    try {
      const result = await this.attemptDeployment(template);
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      if (this.state === 'OPEN') {
        await this.autoRollback();
      }
      throw error;
    }
  }
}
```

```typescript
class TemplateDeploymentStrategy {
  async deployWithBlueGreen(newTemplate: PromptTemplate) {
    // Step 1: Deploy to "green" slot
    await this.deployToSlot('green', newTemplate);
    
    // Step 2: Validate green slot
    const validation = await this.validateSlot('green');
    if (!validation.passed) {
      await this.rollbackSlot('green');
      throw new DeploymentValidationError(validation.errors);
    }
    
    // Step 3: Gradual traffic shift (10% -> 50% -> 100%)
    await this.shiftTraffic('green', 0.1);
    await this.monitorMetrics(Duration.minutes(10));
    
    await this.shiftTraffic('green', 0.5);
    await this.monitorMetrics(Duration.minutes(10));
    
    await this.shiftTraffic('green', 1.0);
    
    // Step 4: Promote green to blue
    await this.promoteSlot('green');
  }
}
```

**Benefits**:
- Zero-downtime deployments
- Instant rollback capability
- Gradual validation with real traffic
- Reduced risk of widespread impact

#### 7. **Design Scalable Template Caching Strategy** (Medium - P2)

**Design Pattern**: Cache-Aside Pattern + Write-Through Cache

```typescript
class TemplateCacheStrategy {
  constructor(
    private redisCache: Redis,
    private templateRepository: Repository<PromptTemplate>,
  ) {}
  
  async getTemplate(roleId: string, version?: string): Promise<PromptTemplate> {
    const cacheKey = this.buildCacheKey(roleId, version);
    
    // Try cache first
    const cached = await this.redisCache.get(cacheKey);
    if (cached) {
      return JSON.parse(cached);
    }
    
    // Cache miss - fetch from database
    const template = await this.templateRepository.findOne({
      where: { roleId, version, isActive: true },
    });
    
    // Write to cache with TTL
    await this.redisCache.setex(
      cacheKey,
      300, // 5 minutes TTL
      JSON.stringify(template),
    );
    
    return template;
  }
  
  async invalidateCache(roleId: string): Promise<void> {
    const pattern = `template:${roleId}:*`;
    const keys = await this.redisCache.keys(pattern);
    if (keys.length > 0) {
      await this.redisCache.del(...keys);
    }
  }
}
```

**Benefits**:
- Reduced database load for frequently accessed templates
- Fast template retrieval for active agents
- Automatic invalidation on deployment
- TTL prevents stale cache issues
